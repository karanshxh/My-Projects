/**
 * MathLib class is a library of simple Math functions. 
 * 
 * @author bhatnagars
 *
 */
public class MathLib {
	/**
	 * Checks if the given number, n is less than max
	 * @param n - number to check
	 * @param max - value to check against
	 * @return true if n is less than max, false otherwise
	 */
	public boolean isLessThan(int n, int max) {
		return n < max;
	}
	/**
	 * Checks if the given number, n is even
	 * @param n - number to check
	 * @return - true if n is even, false otherwise
	 */
	public boolean isEven(int n) {
		return n%2==0;
	}
	/**
	 * Increments the value of n by 1
	 * @param n - value to increment
	 * @return - incremented value
	 */
	public int incrementNum(int n) {
		return n+1;
	}
}
