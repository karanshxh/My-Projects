/**
 * Elevator class simulates an elevator. This elevator can only move 
 * between floors 1 to 20. There is no 13th floor and so
 * every floor after the 13th floor is one less than the actual
 * floor number. For example, if the floor number is 17, the actual
 * floor number is 16.
 * @author bhatnagars
 *
 */
public class Elevator {
	/**
	 * Checks if the floor number is valid, i.e., between 1 and 20
	 * @param floor - floor number to check
	 * @return true if the floor is within the valid range, false otherwise
	 */
	public boolean isOutOfRange(int floor) {
		return (floor <= 0 || floor > 20);
	}
	/**
	 * Checks if the floor number is 13
	 * @param floor - floor number
	 * @return true if the floor number is 13, false otherwise
	 */
	public boolean is13thFloor(int floor) {
		return (floor == 13);
	}
	/**
	 * Checks if the floor number is over 13
	 * @param floor - floor number
	 * @return true if the floor number is over 13, false otherwise
	 */
	public boolean isAbove13 (int floor) {
		return (floor > 13);
	}

}
